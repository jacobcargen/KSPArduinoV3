75void AddInput(bool& referenceToBoolVal, int virtualPin, int debounce = 150)
76{
77 // Increase the size of the array if necessary
78 int newSize = max(numPins + 1, virtualPin + 1);
79 VirtualPin* newPins = new VirtualPin[newSize];
80 // Copy existing inputs to the new array
81 for (int i = 0; i < numPins; i++)
82 {
83 newPins[i] = pins[i];
84 }
85 // Add the new input at the specified index
86 newPins[virtualPin].value = &referenceToBoolVal;
87 newPins[virtualPin].lastState = *newPins[virtualPin].value;
88 newPins[virtualPin].lastDebounceTime = 0;
89 newPins[virtualPin].debounceDelay = debounce;
90 // Delete the old array
91 delete[] pins;
92 // Update to the new array
93 pins = newPins;
94 // Update the number of pins
95 numPins = newSize;
96}
100ButtonState getVirtualPin(int virtualPin, bool waitForChange)
101{
102 if (virtualPin > numPins)
103 return NOT_READY;
104
105 bool inputReading = *pins[virtualPin].value;
106
107 if (((inputReading != pins[virtualPin].lastState) &&
108 (millis() - pins[virtualPin].lastDebounceTime > pins[virtualPin].debounceDelay)))
109 {
110 // Serial.println("Changing last state");
111 pins[virtualPin].lastState = inputReading;
112 pins[virtualPin].lastDebounceTime = millis();
113 return !pins[virtualPin].lastState ? ON : OFF;
114 }
115 if (!waitForChange)
116 {
117 return !pins[virtualPin].lastState ? ON : OFF;
118 }
119 return NOT_READY;
120}

138void _shiftIn(int dataA, int clockEnableA, int clockA, int loadA,
139 int dataB, int clockEnableB, int clockB, int loadB)
140{
141 // Pulse to A
142 digitalWrite(loadA, LOW);
143 delayMicroseconds(5);
144 digitalWrite(loadA, HIGH);
145 delayMicroseconds(5);
146 byte inputA[8];
147 // Get input A data
148 digitalWrite(clockA, HIGH);
149 digitalWrite(clockEnableA, LOW);
150 inputA[0] = shiftIn(dataA, clockA, MSBFIRST);
151 inputA[1] = shiftIn(dataA, clockA, MSBFIRST);
152 inputA[2] = shiftIn(dataA, clockA, MSBFIRST);
153 inputA[3] = shiftIn(dataA, clockA, MSBFIRST);
154 inputA[4] = shiftIn(dataA, clockA, MSBFIRST);
155 inputA[5] = shiftIn(dataA, clockA, MSBFIRST);
156 inputA[6] = shiftIn(dataA, clockA, MSBFIRST);
157 inputA[7] = shiftIn(dataA, clockA, MSBFIRST);
158 digitalWrite(clockEnableA, HIGH);
159
160 for (size_t i = 0; i < 64; i++)
161 {
162 if (i < 8)
163 if (1 == bitRead(inputA[0], i))
164 _sIA[i] = 1;
165 else if (i < 16)
166 if (1 == bitRead(inputA[1], i - 8))
167 _sIA[i] = 1;
168 else if (i < 24)
169 if (1 == bitRead(inputA[2], i - 16))
170 _sIA[i] = 1;
171 else if (i < 32)
172 if (1 == bitRead(inputA[3], i - 24))
173 _sIA[i] = 1;
174 else if (i < 40)
175 if (1 == bitRead(inputA[4], i - 32))
176 _sIA[i] = 1;
177 else if (i < 48)
178 if (1 == bitRead(inputA[5], i - 40))
179 _sIA[i] = 1;
180 else if (i < 56)
181 if (1 == bitRead(inputA[6], i - 48))
182 _sIA[i] = 1;
183 else
184 if (1 == bitRead(inputA[7], i - 56))
185 _sIA[i] = 1;
186 }
187
188 // Pulse to B
189 digitalWrite(loadB, LOW);
190 delayMicroseconds(5);
191 digitalWrite(loadB, HIGH);
192 delayMicroseconds(5);
193 byte inputB[2];
194 // Get input B data
195 digitalWrite(clockB, HIGH);
196 digitalWrite(clockEnableB, LOW);
197 inputB[0] = shiftIn(dataB, clockB, MSBFIRST);
198 inputB[1] = shiftIn(dataB, clockB, MSBFIRST);
199 digitalWrite(clockEnableB, HIGH);
200
201 for (size_t i = 0; i < 16; i++)
202 {
203 if (i < 8)
204 if (1 == bitRead(inputB[0], i))
205 _sIB[i] = 1;
206 else
207 if (1 == bitRead(inputB[1], i - 8))
208 _sIB[i] = 1;
209 }
210}

217void InputClass::init()
218{
219 // Set all states to false
220 testButton = false;
221 testSwitch = false;
222 translationButton = false;
223 rotationButton = false;
224 for (int i = 0; i < 64; i++)
225 {
226 _sIA[i] = false;
227 if (i < 16)
228 _sIB[i] = false;
229 }
230 // Init shift register pins
231 pinMode(SHIFT_IN_A_SERIAL_PIN, INPUT);
232 pinMode(SHIFT_IN_A_CLOCK_PIN, OUTPUT);
233 pinMode(SHIFT_IN_A_CLOCK_ENABLE_PIN, OUTPUT);
234 pinMode(SHIFT_IN_A_LOAD_PIN, OUTPUT);
235 pinMode(SHIFT_IN_B_SERIAL_PIN, INPUT);
236 pinMode(SHIFT_IN_B_CLOCK_PIN, OUTPUT);
237 pinMode(SHIFT_IN_B_CLOCK_ENABLE_PIN, OUTPUT);
238 pinMode(SHIFT_IN_B_LOAD_PIN, OUTPUT);
239 // Init test Inputs
240 pinMode(TEST_SWITCH, INPUT_PULLUP);
241 pinMode(TEST_BUTTON, INPUT_PULLUP);
242 // Init virtual pins
243 initVirtualPins();
244}
